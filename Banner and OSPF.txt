---
- name: Configure a banner on IOS devices
  hosts: ios
  gather_facts: false
  tasks:
    - name: Configure a MOTD banner
      cisco.ios.ios_banner:
        banner: motd
        text: This is an SEIT managed node
        state: present
      notify: ios_save_changes

  handlers:
    - name: ios_save_changes
      cisco.ios.ios_config:
        save_when: always

- name: Enable of Ethernet
  hosts: ios
  gather_facts: false
  tasks:
     -name: Ensure Ethernet2 interface is enabled (no shutdown)
      cisco.ios.ios_interfaces:
        config:
          - name: GigabitEthernet2
            description: "Enabled by SEIT"
            shutdown: false
	    - name: Loopback1
            description: "Enabled by SEIT"
        state: merged

- name: Assign IP address
	  hosts: iosxe1.lab.example.com
	  gather_facts: false
	  tasks:
	    - name: Assign IP address
		cisco.ios.ios_l3_interfaces:
		  config:
		    - name: GigabitEthernet2
			ipv4:
		   	  - address: 10.10.10.1/30
		    - name: Loopback1
			ipv4:
		   	  - address: 172.168.10.5/24
        	   state: merged  

- name: Assign IP address
	  hosts: iosxe2.lab.example.com
	  gather_facts: false
	  tasks:
	    - name: Assign IP address
		cisco.ios.ios_l3_interfaces:
		  config:
		    - name: GigabitEthernet2
			ipv4:
		   	  - address: 10.10.10.2/30
		- name: Loopback1
			ipv4:
		   	  - address: 172.169.10.5/24
        	   state: merged

- name: Configure OSPF
	  hosts: iosxe1.lab.example.com
	  gather_facts: false
	  tasks:
	    - name: Configure OSPF
		cisco.ios.ios_ospfv2:
        	  config:
	          processes:
            	- process_id: 1
              	  network:
	                - address: 10.10.10.0
      	            wildcard_bits: 0.0.0.3
            	      area: 0
                	    - address: 72.168.10.0
	                  wildcard_bits: 0.0.0.255
      	            area: 0
		   state: merged

- name: Configure OSPF
	  hosts: iosxe2.lab.example.com
	  gather_facts: false
	  tasks:
	    - name: Configure OSPF
		cisco.ios.ios_ospfv2:
        	  config:
	          processes:
            	- process_id: 1
              	  network:
	                - address: 10.10.10.0
      	            wildcard_bits: 0.0.0.3
            	      area: 0
                	    - address: 72.169.10.0
	                  wildcard_bits: 0.0.0.255
      	            area: 0
		   state: merged
